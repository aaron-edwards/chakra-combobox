export const Template = (args) => 
  <Checkbox 
    name="Combobox" 
    {...args} 
    filter={(input) => {
      const regexs = input.split(" ").map(part => new RegExp(part, 'i'))
      return (item) => regexs.every(re => item.match(re))
    }}
  />
